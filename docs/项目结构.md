# 项目结构

```
Minecraft-Relic-System/
├── build_jar.bat                         # 一键编译脚本
├── pom.xml                               # Maven 配置
├── README.md                             # 项目说明
├── docs/
│   ├── 开发者文档.md                        # 面向开发者
│   ├── 用户指南.md                          # 面向服主 🆕
│   └── 项目结构.md                          # 本文档
├── src/
│   └── main/
│       ├── java/com/salteddoubao/relicsystem/
│       │   ├── MinecraftRelicSystem.java       # 插件主类
│       │   ├── command/
│       │   │   └── RelicCommand.java           # 命令与Tab补全
│       │   ├── gui/
│       │   │   ├── RelicMainMenuGUI.java       # 主菜单
│       │   │   ├── RelicEquipmentGUI.java      # 装备界面
│       │   │   └── RelicWarehouseGUI.java      # 仓库界面
│       │   ├── listener/
│       │   │   ├── PlayerListener.java         # 玩家事件
│       │   │   ├── RelicGUIListener.java       # GUI交互
│       │   │   └── TreasureBoxListener.java    # 宝箱开启监听 🆕
│       │   ├── manager/
│       │   │   ├── RelicManager.java           # 套装/属性池加载
│       │   │   ├── RelicProfileManager.java    # 档案管理
│       │   │   └── TreasureBoxManager.java     # 宝箱定义与权重抽取 🆕
│       │   ├── relic/
│       │   │   ├── RelicData.java              # 数据模型
│       │   │   ├── RelicSlot.java              # 部位
│       │   │   ├── RelicRarity.java            # 稀有度
│       │   │   ├── RelicStatType.java          # 词条类型
│       │   │   └── PlayerRelicProfile.java     # 玩家档案
│       │   ├── service/
│       │   │   ├── RelicEffectService.java     # 效果应用
│       │   │   ├── StatAggregationService.java # 属性聚合
│       │   │   ├── AttributePlusBridge.java    # AP兼容
│       │   │   └── RelicGenerationService.java # 随机生成与升级 🆕
│       │   ├── storage/
│       │   │   ├── IRelicProfileManager.java
│       │   │   ├── StorageFactory.java
│       │   │   ├── RelicInventoryStorage.java
│       │   │   ├── InventoryProfileManager.java
│       │   │   ├── EnderChestStorage.java
│       │   │   ├── EnderChestProfileManager.java
│       │   │   └── DataMigration.java
│       │   └── util/
│       │       ├── RelicIO.java                # YAML序列化
│       │       ├── RelicItemConverter.java     # 物品↔数据
│       │       └── TreasureBoxItemFactory.java # 宝箱物品工厂 🆕
│       └── resources/
│           ├── plugin.yml                      # 插件描述
│           ├── config.yml                      # 主配置
│           └── relics/
│               ├── sets.yml                    # 套装定义
│               ├── attribute_pool.yml          # 属性池/权重/步进 🆕
│               └── treasure_box.yml            # 宝箱定义/权重/外观 🆕
└── target/                                     # 构建输出
```
