# 项目结构

```
Minecraft-GemRelic/
├── build_jar.bat                         # 一键编译脚本
├── pom.xml                               # Maven 配置
├── README.md                             # 项目说明
├── docs/
│   ├── 用户指南.md                        # 面向服主/玩家的使用说明
│   ├── 开发者文档.md (参见本仓库同目录文件)  # 指向开发文档
│   └── 项目结构.md                        # 本文档（项目结构）
├── src/
│   └── main/
│       ├── java/com/lymc/gemrelic/
│       │   ├── GemRelicPlugin.java          # 插件主类
│       │   ├── command/
│       │   │   └── RelicCommand.java        # 命令
│       │   ├── gui/
│       │   │   ├── RelicMainMenuGUI.java    # 主菜单
│       │   │   ├── RelicEquipmentGUI.java   # 装备界面
│       │   │   └── RelicWarehouseGUI.java   # 仓库界面
│       │   ├── listener/
│       │   │   ├── PlayerListener.java      # 玩家事件
│       │   │   └── RelicGUIListener.java    # GUI交互
│       │   ├── manager/
│       │   │   ├── RelicManager.java        # 套装管理
│       │   │   └── RelicProfileManager.java # YAML档案管理（兼容）
│       │   ├── storage/
│       │   │   ├── IRelicProfileManager.java     # 存储接口
│       │   │   ├── StorageFactory.java           # 存储工厂
│       │   │   ├── RelicInventoryStorage.java    # 独立存储（v3，2000容量）
│       │   │   ├── InventoryProfileManager.java  # 独立存储档案适配器
│       │   │   ├── EnderChestStorage.java        # 末影箱存储（备用）
│       │   │   ├── EnderChestProfileManager.java # 末影箱档案（备用）
│       │   │   └── DataMigration.java            # 数据迁移
│       │   ├── relic/
│       │   │   ├── RelicData.java           # 数据模型
│       │   │   ├── RelicSlot.java           # 部位枚举
│       │   │   ├── RelicRarity.java         # 稀有度
│       │   │   ├── RelicStatType.java       # 词条类型
│       │   │   └── PlayerRelicProfile.java  # 玩家档案接口
│       │   ├── service/
│       │   │   ├── RelicEffectService.java  # 套装效果
│       │   │   ├── StatAggregationService.java # 属性聚合
│       │   │   └── AttributePlusBridge.java # AP兼容
│       │   └── util/
│       │       ├── RelicIO.java             # YAML序列化
│       │       └── RelicItemConverter.java  # 物品↔数据转换
│       
│       └── resources/
│           ├── plugin.yml                   # 插件描述
│           ├── config.yml                   # 主配置
│           └── relics/
│               └── sets.yml                 # 套装定义
└── target/                                  # 构建输出
```
